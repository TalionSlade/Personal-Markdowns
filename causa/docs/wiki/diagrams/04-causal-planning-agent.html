Loop body (CausalPlanningAgent)

for step_index in range(self._step_budget):
    ctx        = self._snapshot_context(step_index=step_index, state=state)         # 1
    choice     = self._choose_action(candidates, context=ctx, task=task)             # 2: DoWhyActionScorer
    tool       = self._tools[choice.action.name]                                     # 5
    output     = tool(ToolInput(args={}), state=state)
    obs, outcome = self._record_observation(state=state, choice=choice,             # 6
                                            tool_output=output)
    self._after_step(step=step, task=task, state=state)                              # 3+7: history append, reflect
    trace.append(step)
    if outcome >= self._success_threshold: break                            # SOLVED
    state = self._next_state(prev_state=state, observation=obs)                      # carry mediators forward
